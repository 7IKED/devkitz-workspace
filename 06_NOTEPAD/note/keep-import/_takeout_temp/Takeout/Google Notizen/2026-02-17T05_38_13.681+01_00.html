// IndexedDB helper for user folders + media URLs
function openDb() {
return new Promise((resolve, reject) => {
if (db) return resolve(db);
const req = indexedDB.open('lp_userdata_v1', 1);
req.onupgradeneeded = (e) => {
const idb = e.target.result;
if (!idb.objectStoreNames.contains('users')) {
const os = idb.createObjectStore('users', { keyPath: 'id', autoIncrement: true });
os.createIndex('byName', 'name', { unique: false });
}
if (!idb.objectStoreNames.contains('media')) {
const ms = idb.createObjectStore('media', { keyPath: 'id', autoIncrement: true });
ms.createIndex('byUser', 'userId', { unique: false });
}
};
req.onsuccess = (e) => { db = e.target.result; resolve(db); };
req.onerror = (e) => reject(e);
});
}

async function saveUserProfile(profile) {
const idb = await openDb();
return new Promise((res, rej) => {
const tx = idb.transaction('users', 'readwrite');
const store = tx.objectStore('users');
const req = store.put(profile);
req.onsuccess = () => res(req.result);
req.onerror = (e) => rej(e);
});
}

async function addMediaForUser(userId, media) {
const idb = await openDb();
return new Promise((res, rej) => {
const tx = idb.transaction('media', 'readwrite');
const store = tx.objectStore('media');
const req = store.add({ userId, ...media });
req.onsuccess = () => res(req.result);
req.onerror = (e) => rej(e);
});
}

// Utility: UID
function uid() { return 'f_'+Math.random().toString(36).slice(2,9); }

// Load favorites from storage
function loadFavorites() {
chrome.storage.local.get({ lpFavorites: [] }, (items) => {
favorites = items.lpFavorites || [];
renderFavorites();
});
}

function saveFavorites() {
chrome.storage.local.set({ lpFavorites: favorites });
}

function renderFavorites() {
favListEl.innerHTML = '';
favorites.forEach(f => {
const div = document.createElement('div');
div.className = 'fav-item';
div.innerHTML = `<div>
<strong>${escapeHtml(f.name || f.url)}</strong><div style="font-size:12px;opacity:0.7">${escapeHtml(f.url)}</div>
</div>
<div>
<button data-id="${f.id}" class="openFav">Öffnen</button>
<button data-id="${f.id}" class="toggleFav">${f.enabled ? '✔' : '✖'}</button>
<button data-id="${f.id}" class="delFav">🗑</button>
</div>`;
favListEl.appendChild(div);
});

});
$$('.delFav').forEach(b => b.onclick = () => {

function addFavorite(url, name) {
const entry = { id: uid(), url, name: name || url, enabled: true, folderId: null };
favorites.push(entry);
saveFavorites();
renderFavorites();
log(`Favorit hinzugefügt: ${entry.name}`);
}

function removeFavorite(id) {
favorites = favorites.filter(f => f.id !== id);
saveFavorites();
renderFavorites();
}

function toggleFavoriteEnabled(id) {
const f = favorites.find(x => x.id === id);
if (!f) return;
f.enabled = !f.enabled;
saveFavorites();
renderFavorites();
}

// open favourite in tab (mobile/desktop scaling)
function openFavoriteInTab(id) {
const f = favorites.find(x => x.id === id);
if (!f) return;
const mobile = mobileModeChk.checked;
const scale150 = scale150Chk.checked;
let url = f.url;
if (mobile) {
url = toMobileUrl(url);
}
chrome.tabs.create({ url }, (tab) => {
log(`Tab geöffnet: ${f.name}`);
// optionally inject content script to collect profile metadata once loaded
setTimeout(() => {
tryInjectCollector(tab.id);
}, 2000);
});
}

function toMobileUrl(url) {
// naive mobile conversion heuristics (can be improved per site)
if (/stripchat\.com/.test(url)) return url + (url.includes('?') ? '&' : '?') + 'mobile=true';
if (/chaturbate\.com/.test(url)) return url.replace('chaturbate.com', 'm.chaturbate.com');
return url;
}

// Try to inject a collector script into tab to scrape profile metadata & media urls
function tryInjectCollector(tabId) {
// run content-script collector which will postMessage back with user info
chrome.scripting.executeScript({
target: { tabId },
func: () => {
// Collector runs in page context — gather basic metadata and media URLs
const metadata = {};
metadata.title = document.title || '';
// profile name heuristics
const nameEl = document.querySelector('.profile-username') || document.querySelector('.username') || document.querySelector('h1');
metadata.name = nameEl ? nameEl.innerText.trim() : '';
metadata.media = [];
document.querySelectorAll('img, video source, video').forEach(el => {
try {
if (el.tagName === 'IMG' && el.src) metadata.media.push({ type: 'image', url: el.src });
if (el.tagName === 'VIDEO') {
if (el.currentSrc) metadata.media.push({ type: 'video', url: el.currentSrc });
el.querySelectorAll('source').forEach(s => s.src && metadata.media.push({ type: 'video', url: s.src }));
}
if (el.tagName === 'SOURCE' && el.src) metadata.media.push({ type: 'video', url: el.src });
} catch (e) {}
});
// send to extension via window.postMessage
window.postMessage({ lpCollector: true, metadata }, '*');
}
}).catch(e => {
log('Collector injection failed: ' + (e.message || e));
});
}

// Radar logic: open tabs for enabled favorites and monitor their status.
function startRadar() {
if (radarRunning) return;
radarRunning = true;
radarToggleBtn.classList.remove('grey'); radarToggleBtn.classList.add('green');
radarToggleBtn.textContent = 'Radar läuft';
statusGlobal.className = 'status green'; statusGlobal.textContent = 'Radar aktiv';

// For each enabled favorite, create a background tab (non-active) and monitor it
favorites.filter(f=>f.enabled).forEach(async f => {
chrome.tabs.create({ url: f.url, active: false }, (tab) => {
monitoringTabs[f.id] = { tabId: tab.id, status: 'loading', lastSeen: Date.now() };
renderRadarList();
// monitor tab updates
});
});
}

function stopRadar() {
radarRunning = false;
radarToggleBtn.classList.remove('green'); radarToggleBtn.classList.add('grey');
radarToggleBtn.textContent = 'Radar starten';
statusGlobal.className = 'status grey'; statusGlobal.textContent = 'Offline';
// close monitoring tabs
Object.values(monitoringTabs).forEach(m => {
try { chrome.tabs.remove(m.tabId); } catch(e){}
});
monitoringTabs = {};
renderRadarList();
}

}

// Message listener from injected collector
window.addEventListener('message', async (ev) => {
const msg = ev.data;
if (!msg || !msg.lpCollector) return;
const meta = msg.metadata;
if (!meta) return;
// Create or update user profile in IndexedDB

const userId = await saveUserProfile(profile);
log(`Profil gespeichert: ${profile.name} (id ${userId})`);
// Save media entries

catch(e){}
}
});

// Collect media for a given open tab (attempt fetch & store as blob) — note: may fail due to CORS
async function collectMediaFromTab(tabId, userId) {
// execute script to return media URLs
try {
const res = await chrome.scripting.executeScript({
target: { tabId },
func: () => {
const out = [];
document.querySelectorAll('img, video, source').forEach(el => {
try {
const url = el.src || el.currentSrc || el.getAttribute('src');
if (url) out.push(url);
} catch(e){}
});
return out;
}
});
const urls = res?.[0]?.result || [];
for (const u of urls) {
try {
await addMediaForUser(userId, { url: u, type: (u.match(/\.(mp4|webm|ogg)/) ? 'video' : 'image'), capturedAt: Date.now() });
} catch (e) { console.warn('media add fail', e); }
}
log(`Medien-URLs gesammelt (${urls.length})`);
return urls;
} catch (e) {
log('Medien-Abfrage fehlgeschlagen: ' + e.message);
return [];
}
}

// Export session as markdown (.md) -> gather favorites, monitoring state, users, and logs (uses storage + IndexedDB)
async function exportSessionMarkdown() {
const now = new Date();
let md = `# Chat-Protokoll (Session)\nDatum: ${now.toLocaleString('de-DE')}\n\n`;
// favorites summary
md += `## Favoriten (${favorites.length})\n`;
favorites.forEach(f => {
const mon = monitoringTabs[f.id];
const status = mon ? mon.status : 'stopped';
md += `- **${f.name}** — ${f.url} — Status: ${status}\n`;
});

// User summary from IndexedDB (quick)
const idb = await openDb();
const users = await new Promise((res, rej) => {
const tx = idb.transaction('users', 'readonly');
const store = tx.objectStore('users');
const req = store.getAll();
req.onsuccess = () => res(req.result || []);
req.onerror = e => rej(e);
});
md += `\n## Gespeicherte Profile (${users.length})\n`;
users.forEach(u => {
md += `- ${u.name} — id:${u.id}\n`;
});

// Chat logs: if you previously stored chat messages in chrome.storage (from content-script), include them
const stored = await new Promise(r => chrome.storage.local.get(['lpChatLog'], it => r(it.lpChatLog || [])));
md += `\n## Chatverlauf (${stored.length} Nachrichten)\n`;
stored.forEach(m => {
md += `- [${m.time}] **${m.user}**: ${m.text} (${m.lang || 'Unbekannt'})\n`;
});

// Totals (if you track tokens in storage)
const totals = await new Promise(r => chrome.storage.local.get(['lpTotals'], it => r(it.lpTotals || { tokens:0, euros:0 })));
md += `\n## Session-Totals\n- Tokens: ${totals.tokens}\n- Euro: ${totals.euros.toFixed(2)} €\n`;

// Copy + download
await navigator.clipboard.writeText(md).catch(()=>{ log('Clipboard Copy failed'); });
// create blob download
const blob = new Blob([md], { type: 'text/markdown' });
const url = URL.createObjectURL(blob);
chrome.downloads?.download ? chrome.downloads.download({ url, filename: `lp-session-${Date.now()}.md` }) : (() => {
const a = document.createElement('a'); a.href = url; a.download = `lp-session-${Date.now()}.md`; a.click(); URL.revokeObjectURL(url);
})();
log('Session exportiert (.md) und kopiert');
}
